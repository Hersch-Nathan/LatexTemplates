% Playscript Class Test File
% This file demonstrates all features of playscript.cls with updated syntax
% Based on R.U.R. (Rossum's Universal Robots) by Karel Čapek

\documentclass{../playscript}

% Title page information
\title{R. U. R.}
\subtitle{(Rossum's Universal Robots)}
\author{KAREL ČAPEK}
\translatedby{Translated by Paul Selver and Nigel Playfair}
\performancenote{%
Robots of the world! The power of man has fallen!\\
A new world has arisen: the Rule of the Robots!\\
March!%
}

\begin{document}

% Test Case 1: Title Page
\maketitle

% Test Case 2: Character List
\begin{characters}
\character{HARRY DOMIN}
\character{SULLA}
\character{HELENA GLORY}
\end{characters}

% Test Case 3: Act Heading with Scene Description
\act{I}

% Test Case 4: Scene Heading
\scene{Central office of the factory}

% Test Case 5: Scene Description
\begin{scenedesc}
Entrance on the right. The windows look out on factory chimneys. DOMIN sits at a large writing table. SULLA is typing letters at a desk near the windows.
\end{scenedesc}

% Test Case 6: Basic Dialogue with Parenthetical
\speaker{DOMIN}
\paren{dictating}
\dialogue{Ready?}

\speaker{SULLA}
\dialogue{Yes.}

% Test Case 7: Long Dialogue
\speaker{DOMIN}
\dialogue{To E. M. McVicker and Co., Southampton, England. "We undertake no guarantee for goods damaged in transit."}

% Test Case 8: Stage Direction
\stagedirection{SULLA types the letter}

\speaker{DOMIN}
\dialogue{Ready?}

\speaker{SULLA}
\dialogue{Yes.}

% Test Case 9: Multiple exchanges
\speaker{DOMIN}
\dialogue{Where's that letter from Valparaiso?}

\speaker{SULLA}
\dialogue{Here.}

\stagedirection{DOMIN takes the letter}

\speaker{DOMIN}
\paren{reading}
\dialogue{Good. That's done then.}

% Test Case 10: Longer stage direction
\begin{stage}
A knock at the door. MARIUS enters with HELENA. MARIUS exits. HELENA looks around.
\end{stage}

\speaker{DOMIN}
\dialogue{How do you do, madame?}

\speaker{HELENA}
\dialogue{How do you do?}

\speaker{DOMIN}
\dialogue{What can I do for you?}

\speaker{HELENA}
\dialogue{You are Mr. Domin, the General Manager?}

\speaker{DOMIN}
\dialogue{I am.}

\speaker{HELENA}
\paren{hesitant}
\dialogue{I have come because I believe you can help me.}

% Test Case 11: Extended monologue
\speaker{DOMIN}
\dialogue{It will give me great pleasure. We here at Rossum's Universal Robots are always happy to assist visitors. This factory represents old Rossum's vision of creating artificial workers to free humanity from labor.}

% Test Case 12: Rapid exchange
\speaker{HELENA}
\dialogue{But surely they must have feelings?}

\speaker{DOMIN}
\dialogue{No feelings.}

\speaker{HELENA}
\dialogue{No soul?}

\speaker{DOMIN}
\dialogue{No soul. They are simply living machines.}

\curtain

% Test Case 13: Act II
\act{II}

\scene{Ten years later}

\begin{scenedesc}
The office looks similar but the maps show more extensive routes. DOMIN stands at the window. HELENA enters.
\end{scenedesc}

\speaker{HELENA}
\dialogue{You wanted to see me?}

\speaker{DOMIN}
\dialogue{Yes. I have bad news.}

\speaker{HELENA}
\paren{alarmed}
\dialogue{What's happened?}

\speaker{DOMIN}
\dialogue{The Robots are organizing. They're making demands.}

\speaker{HELENA}
\dialogue{How is that possible?}

\speaker{DOMIN}
\dialogue{Something has changed in them. They're something more than machines now.}

\fadeout

\vspace{2em}

\begin{center}
\textsc{End of Play}
\end{center}

\end{document}
